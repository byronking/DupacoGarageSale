ALTER PROCEDURE [dbo].[DeleteGarageSale]

@sale_id int

as

begin

set nocount on;

	begin
		
		delete from GarageSaleItems where sale_id = @sale_id;
		delete from GarageSaleDatesTimes where sale_id = @sale_id;
		delete from GarageSales where sale_id = @sale_id;

	end
end
