create PROCEDURE [dbo].[GetGarageSaleSpecialItems]

@sale_id int

as

begin

set nocount on;

	begin
		
		select special_items_id, title, description, picture_link, price, sale_id, item_subcategory_id, item_subcategory_id
			from SpecialItems where sale_id = @sale_id

	end
end
